  ```java
  class A{
      i;
      ma()
  }
  class B extends A{
      j;
      mb()
  }
  
  A a2 = new B();  // 얘는 자동 형변환
  a2.i; // 형변환 가능( A거라서 가능)
  a2.j; // 형변환 불가능
  a2.ma(); //형변환 가능
  a2.mb(); // 형변환 불가능
  
  // 명시적 형변환
  A a2 = new B();

  a2.i; // 형변환 가능( A거라서 가능)
  a2.j; // 명시적 변환 가능
  a2.ma(); //형변환 가능
  a2.mb(); // 명시적 변환 가능
  ```
